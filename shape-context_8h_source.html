99

107  cyprime = -factor * ry * x1prime / rx,
108  cx = cxprime + (x1 + x2) / 2, cy = cyprime + (y1 + y2) / 2,
109  start_theta = -atan2((y1 - cy) * rx, (x1 - cx) * ry),
